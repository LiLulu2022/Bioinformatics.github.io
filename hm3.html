bash.sh脚本内容：

#!/bin/bash

folder="bash_homework"
if [ ! -d "$folder" ]; then
  echo "文件夹 $folder 不存在"
  exit 1
fi

touch filenames.txt
touch dirname.txt

> filenames.txt
> dirname.txt

for item in "$folder"/*; do
  if [ -f "$item" ]; then
    echo "$(basename "$item")" >> filenames.txt
  elif [ -d "$item" ]; then
    echo "$(basename "$item")" >> dirname.txt
  fi
done

echo "文件名已写入 filenames.txt"
echo "子文件夹名已写入 dirname.txt"
:wq
